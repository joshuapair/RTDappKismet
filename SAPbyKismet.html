  <script>

    function updateCount() {

      const selected = document.querySelectorAll(".send-button.selected").length;

      document.getElementById("selectedCount").textContent = `Selected: ${selected}`;

    }

    function toggleSelect(button) {

      button.classList.toggle("selected");

      updateCount();

    }

    function clearSelection() {

      document.querySelectorAll(".send-button.selected").forEach(btn => btn.classList.remove("selected"));

      updateCount();

    }

    function buildCombinedMessage() {

      const selectedButtons = Array.from(document.querySelectorAll(".send-button.selected"));

      if (selectedButtons.length === 0) return null;

      const lines = [];

      lines.push("Here are the requested links/info:");

      lines.push("");

      selectedButtons.forEach((btn, idx) => {

        const title = btn.getAttribute("data-title") || `Item ${idx + 1}`;

        const link = btn.getAttribute("data-link") || "";

        const customBody = btn.getAttribute("data-body"); // optional

        lines.push(`— ${title}`);

        if (customBody && customBody.trim().length > 0) {

          lines.push(customBody.trim());

          lines.push("");

          lines.push(`Link: ${link}`);

        } else {

          lines.push(link);

        }

        lines.push(""); // blank line between items

      });

      return {

        subject: `Links (${selectedButtons.length})`,

        body: lines.join("\n")

      };

    }

    function sendSelected() {

      const recipient = document.getElementById("recipient").value.trim();

      if (!recipient) {

        alert("Please enter an email or phone number.");

        return;

      }

      const payload = buildCombinedMessage();

      if (!payload) {

        alert("Select at least one item first.");

        return;

      }

      // Basic routing: email if it contains "@", otherwise SMS

      if (recipient.includes("@")) {

        const mailto = `mailto:${recipient}?subject=${encodeURIComponent(payload.subject)}&body=${encodeURIComponent(payload.body)}`;

        window.open(mailto);

      } else {

        // SMS (note: desktop browsers may not support sms: well)

        const sms = `sms:${recipient}?body=${encodeURIComponent(payload.body)}`;

        window.open(sms);

      }

    }

    // init

    updateCount();
